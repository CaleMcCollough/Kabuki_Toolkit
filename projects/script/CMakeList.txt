# Collect sources into the variable KABUKI_SOURCES without
# having to explicitly list each header and source file.
#
# CMake documentation states "We do not recommend using GLOB to collect a
# list of source files from your source tree. If no CMakeLists.txt file
# changes when a source is added or removed then the generated build system
# cannot know when to ask CMake to regenerate".
file (GLOB KABUKI_SOURCES
      "*.h")
 
# Collect sources into the variable KABUKI_TOOLKIT_SOURCES
file (GLOB KABUKI_TOOLKIT_SOURCES
      "include/*.h"
      "impl/*.cc")
 
# The recommended way to collect sources in variable 
# CHINESE_ROOM_SOURCES by explicitly specifying the source files
set  (CHINESE_ROOM_SOURCES
      "chinese_room/AdvancedFunctions.h"
      "chinese_room/AdvancedFunctions.cc")
 
# Create named folders for the sources within the .vcproj
# Empty name lists them directly under the .vcproj
source_group("" FILES ${KABUKI_SOURCES})
source_group("kabuki_toolkit" FILES ${KABUKI_TOOLKIT_SOURCES})
source_group("chinese_room" FILES ${CHINESE_ROOM_SOURCES})
 
# Properties->C/C++->General->Additional Include Directories
include_directories (.)
 
# Set Properties->General->Configuration Type to Dynamic Library(.dll)
# Creates kabuki.dll with the listed sources collected in the variables
# Also adds sources to the Solution Explorer
add_library(kabuki SHARED ${KABUKI_SOURCES} 
                        ${KABUKI_TOOLKIT_SOURCES} 
                        ${CHINESE_ROOM_SOURCES})
 
# Creates folder "libraries" and adds result project (kabuki.vcproj)
set_property(TARGET kabuki PROPERTY FOLDER "libraries")
 
# Properties->General->Output Directory
set_result_properties(kabuki PROPERTIES
                      RUNTIME_OUTPUT_DIRECTORY ${PROJECT_BINARY_DIR}/bin)
 
# Adds logic to INSTALL.vcproj to copy kabuki.dll to the destination directory
install (TARGETS kabuki
         RUNTIME DESTINATION ${PROJECT_SOURCE_DIR}/_install
         LIBRARY DESTINATION ${PROJECT_SOURCE_DIR}/_install)
